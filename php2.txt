<?php
if (isset($_GET['choice'], $_GET['filename'])) {
    $choice = (int)$_GET['choice'];   
    $file   = trim($_GET['filename']);
    $txt    = $_GET['content'] ?? "";

    switch ($choice) {
        case 1:
            $handle = fopen($file, "w");
            if ($handle) {
                fclose($handle);
                echo "<p>File '$file' created successfully.</p>";
            } else {
                echo "<p>File creation failed.</p>";
            }
            break;

        case 2: 
            if (file_put_contents($file, $txt) !== false) {
                echo "<p>Content written to '$file'.</p>";
            } else {
                echo "<p>Failed to write to file.</p>";
            }
            break;

        case 3: 
            if (file_exists($file)) {
                echo "<h3>Contents of '$file':</h3>";
                echo "<pre>" . htmlspecialchars(file_get_contents($file)) . "</pre>";
            } else {
                echo "<p>File not found.</p>";
            }
            break;

        case 4: 
            $newFile = "copy_" . $file;
            if (file_exists($file) && copy($file, $newFile)) {
                echo "<p>File copied to '$newFile'.</p>";
            } else {
                echo "<p>Copy failed.</p>";
            }
            break;

        case 5: // Delete
            if (file_exists($file) && unlink($file)) {
                echo "<p>File '$file' deleted successfully.</p>";
            } else {
                echo "<p>Delete failed.</p>";
            }
            break;

        default:
            echo "<p>Invalid choice (must be 1â€“5).</p>";
    }
}
?>
