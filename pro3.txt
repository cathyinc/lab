<h2>Count Uppercase and Lowercase Letters</h2>
<form method="get">
    Enter a string:<br>
<input type="text" name="input"><br><br>
    Enter your choice (1 or 2):<br>
<input type="text" name="choice"><br><br>
<input type="submit" value="Submit">
</form>
<?php
if ($_GET["input"] && $_GET["choice"]) {
    $str = $_GET["input"];
    $choice = $_GET["choice"];
    $uppercaseCount = 0; $lowercaseCount = 0;
    for ($i = 0; $i<strlen($str); $i++) {
        if (ctype_upper($str[$i])) $uppercaseCount++;
        if (ctype_lower($str[$i])) $lowercaseCount++;
    }
    echo $choice == "1" ? "Uppercase: $uppercaseCount" : 
        ($choice == "2" ? "Lowercase: $lowercaseCount" : 
        "Invalid choice");
}
?>
