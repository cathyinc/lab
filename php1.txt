<!DOCTYPE html>

<body>
<form method="get">
    <input type="radio" name="choice" value="1"> Sum of Digits<br>
    <input type="radio" name="choice" value="2"> Factorial<br><br>
    <input type="text" name="b" placeholder="Enter a number"><br><br>
    <input type="submit" value="Submit">
</form>

<?php
if (isset($_GET["choice"]) && isset($_GET["b"]) && $_GET["b"] !== "") {
    $n = (int)$_GET["b"];
    $choice = (int)$_GET["choice"];

    switch ($choice) {
        case 1:
            $digits = str_split((string)abs($n));
            $sum = array_sum($digits);
            echo "<p>Sum of digits of $n is: $sum</p>";
            break;

        case 2:
            if ($n < 0) {
                echo "<p>Factorial not defined for negative numbers.</p>";
            } else {
                $factorial = 1;
                for ($i = 1; $i <= $n; $i++) {
                    $factorial *= $i;
                }
                echo "<p>Factorial of $n is: $factorial</p>";
            }
            break;

        default:
            echo "<p>Invalid choice</p>";
    }
}
?>
